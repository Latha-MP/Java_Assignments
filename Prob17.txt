

package Assignments.Prob17;

public class Prog17 {

    public static void main(String[] args) {
        for (int i = 0; i < 100; i++) {
            if(i%2==0)
            {
                System.out.println(i+" Even number");
            }
            else
            {
                System.out.println(i+" Odd number");
            }
        }
    }
}
